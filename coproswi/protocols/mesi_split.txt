(define_coherency_manager
   (add_state stable MEM_M)
   (add_state stable MEM_S)
   (add_state stable MEM_I)

   (set_default_state MEM_I)

   (add_state transient MEM_I_D)
   (add_state transient MEM_I_O_S_B)
   (add_state transient MEM_S_D)

   (MEM_I
      (CMD_CCTRL_GET_SHARED
         (read_data)
         (sender!CMD_DATA_EXCLUSIVE_MSG)
         (store_owner)
         (set_state MEM_M)
      )
      (CMD_CCTRL_GET_MODIFIED
         (read_data)
         (sender!CMD_DATA_MSG)
         (store_owner)
         (set_state MEM_M)
      )
      (CMD_CCTRL_PUT_MODIFIED
         (if_is_owner (none) (ignore))
      )

      (CMD_DATA_MSG (none))
      (CMD_DATA_EXCLUSIVE_MSG (none))
   )

   (MEM_M
      (CMD_CCTRL_GET_SHARED
         (reset_owner)
         (set_state MEM_S_D)
      )
      (CMD_CCTRL_GET_MODIFIED
         (store_owner)
      )
      (CMD_CCTRL_PUT_MODIFIED
         (if_is_owner
            (
               (reset_owner)
               (set_state MEM_I_D)
            )
            (ignore)
         )
      )

      (CMD_DATA_MSG
         (write_data)
         (resume)
         (set_state MEM_I_O_S_B)
      )
      (CMD_DATA_EXCLUSIVE_MSG (none))
   )

   (MEM_I_D
      (CMD_CCTRL_GET_SHARED (stall))
      (CMD_CCTRL_GET_MODIFIED (stall))

      (CMD_CCTRL_PUT_MODIFIED
         (if_is_owner (none) (ignore))
      )

      (CMD_DATA_MSG
         (write_data)
         (resume)
         (set_state MEM_I)
      )
      (CMD_DATA_EXCLUSIVE_MSG (resume) (set_state MEM_I))
   )

   (MEM_S_D
      (CMD_CCTRL_GET_SHARED (stall))
      (CMD_CCTRL_GET_MODIFIED (stall))

      (CMD_CCTRL_PUT_MODIFIED
         (if_is_owner (none) (ignore))
      )

      (CMD_DATA_MSG
         (write_data)
         (resume)
         (set_state MEM_S)
      )
      (CMD_DATA_EXCLUSIVE_MSG
         (resume)
         (set_state MEM_S)
      )
   )

   (MEM_I_O_S_B
      (CMD_CCTRL_GET_SHARED
         (reset_owner)
         (set_state MEM_S)
      )
      (CMD_CCTRL_GET_MODIFIED
         (store_owner)
        ;; (set_state MEM_M)
      )

      (CMD_CCTRL_PUT_MODIFIED
         (if_is_owner
            (
               (reset_owner)
               (set_state MEM_I)
            )
            (ignore)
         )
      )

      (CMD_DATA_MSG (none))
      (CMD_DATA_EXCLUSIVE_MSG (none))
   )

   (MEM_S
      (CMD_CCTRL_GET_SHARED
         (read_data)
         (sender!CMD_DATA_MSG)
      )
      (CMD_CCTRL_GET_MODIFIED
         (read_data)
         (sender!CMD_DATA_MSG)
         (store_owner)
         (set_state MEM_M)
      )

      (CMD_CCTRL_PUT_MODIFIED
         (if_is_owner (none) (ignore))
      )

      (CMD_DATA_MSG (none))
      (CMD_DATA_EXCLUSIVE_MSG (none))
   )
)
